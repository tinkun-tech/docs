    AMOE Module¶

    This guide covers the AMOE (Alternate Method Of Entry) module for bulk code assignment and email distribution.

    Overview¶

    The AMOE module allows you to:

    • Assign codes to users in bulk via email list
    • Send codes via email with multi-language support
    • Manage assignments through Django admin
    • Track email delivery status

    AMOE Model¶

    Create a model that inherits from AbstractAMOECode:

    Basic Implementation¶

    # myapp/models.py
from tinkun_code.models.abstract import AbstractCode, AbstractAMOECode

class Code(AbstractCode):
    pass

class AMOECode(AbstractAMOECode):
    pass

    With Tracking Fields¶

    # myapp/models.py
from django.db import models

from tinkun_code.models.abstract import AbstractCode, AbstractAMOECode

class Code(AbstractCode):
    campaign = models.CharField(max_length=100, blank=True)

class AMOECode(AbstractAMOECode):
    email_sent_at = models.DateTimeField(null=True, blank=True)
    batch_id = models.CharField(max_length=50, blank=True)
    language = models.CharField(max_length=2, default='en')

    def __str__(self):
        return f"AMOE-{self.code.code}"

    AbstractAMOECode Fields¶

    Field Type Description
    code OneToOneField Reference to Code
    user ForeignKey Assigned user
    created_at DateTimeField When created

    Email Service Implementation¶

    You must implement an email service for sending codes

    Configuration¶

    Basic AMOE Setup¶

    # settings.py
TINKUN_CODE = {
    "CODE_MODEL": "myapp.Code",
    "AMOE_CODE_MODEL": "myapp.AMOECode",
    "AMOE_CODE_EMAIL_SERVICE": "myapp.services.EmailService",
}

    With Language Support¶

    # myapp/models.py (User model)
class User(AbstractUser):
    language = models.CharField(max_length=2, default='en')

# settings.py
TINKUN_CODE = {
    "CODE_MODEL": "myapp.Code",
    "AMOE_CODE_MODEL": "myapp.AMOECode",
    "AMOE_CODE_EMAIL_SERVICE": "myapp.services.EmailService",
    "DEFAULT_LANGUAGE": "es",  # Default when user language not available
    "MAX_EMAILS_PER_BATCH": 500,  # Limit batch size
}

    Admin Interface¶

    AMOE Admin¶

    # myapp/admin.py
from django.contrib import admin

from tinkun_code.admin import AMOECodeAdminMixin

from .models import AMOECode

@admin.register(AMOECode)
class AMOECodeAdmin(AMOECodeAdminMixin, admin.ModelAdmin):
    list_display = ['code', 'user', 'email_sent_at', 'created_at']
    list_filter = ['language', 'batch_id']
    search_fields = ['code__code', 'user__email']
    readonly_fields = ['email_sent_at', 'created_at', 'updated_at']

    Admin Features¶

    The AMOECodeAdminMixin provides:

    • Bulk assignment interface in admin
    • Email sending form with textarea for email list
    • Assignment tracking and results page
    • Optimized queries for large datasets

    Email Service Examples¶

    One language¶

    # myapp/services.py
from typing import List

from tinkun_code.interfaces.amoe_code_service_interface import AbstractAMOECodeEmailService

class AMOECodeEmailService(AbstractAMOECodeEmailService):
    email_service = EmailService()

    def send_email_batch(self, email_list: List[str], context_list: List[dict], language: str):
        batch_number = self.email_service.create_batch_email_audits(
            email_list=email_list,
            email_type=EmailTypes.AMOE_CODE,
            source="amoe_code_service",
        )

        self.email_service.send_email_batch(
            email_list=email_list,
            email_type=EmailTypes.AMOE_CODE,
            filename=AMOE_CODE_EMAIL_FILENAME,
            user_specific_context_list=context_list,
            batch_number=batch_number,
            subject=AMOE_CODE_EMAIL_SUBJECT,
        )

    Multiple language support¶

    # myapp/services.py
from typing import List

from tinkun_code.interfaces.amoe_code_service_interface import AbstractAMOECodeEmailService

class AMOECodeEmailService(AbstractAMOECodeEmailService):
    email_service = EmailService()

    def send_email_batch(self, email_list: List[str], context_list: List[dict], language: str):
        batch_number = self.email_service.create_batch_email_audits(
            email_list=email_list,
            email_type=EmailTypes.AMOE_CODE,
            source="amoe_code_service",
        )

        filename, subject = AMOE_TEMPLATES_LANGUAGE_MAP[language]

        self.email_service.send_email_batch(
            email_list=email_list,
            email_type=EmailTypes.AMOE_CODE,
            filename=filename,
            user_specific_context_list=context_list,
            batch_number=batch_number,
            subject=subject,
        )
